<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Visualizar Dados</title>
    <style>
        table {
            width: 100%;
            border-collapse: collapse;
            margin-top: 16px;
        }
        th, td {
            border: 1px solid #ddd;
            padding: 8px;
            text-align: left;
        }
        th {
            background-color: #f4f4f4;
        }
    </style>
    <script>
        async function buscarDados() {
            const matricula = document.getElementById('matricula').value;
            const response = await fetch(`../scripts/obter_viatura.php?Matricula=${matricula}`);
            const viatura = await response.json();
            
            // Exibir os dados em tabela
            const resultado = document.getElementById('resultado');
            resultado.innerHTML = ''; // Limpar resultados anteriores

            if (viatura.length === 0) {
                resultado.innerHTML = '<p>Nenhum dado encontrado.</p>';
                return;
            }

            const table = document.createElement('table');
            const thead = document.createElement('thead');
            const tbody = document.createElement('tbody');

            thead.innerHTML = `
                <tr>
                    <th>Nome</th>
                    <th>Marca</th>
                    <th>Modelo</th>
                    <th>Ano</th>
                    <th>Cor</th>
                </tr>
            `;
            table.appendChild(thead);

            viatura.forEach(item => {
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td>${item.nome}</td>
                    <td>${item.marca}</td>
                    <td>${item.modelo}</td>
                    <td>${item.ano}</td>
                    <td>${item.cor}</td>
                `;
                tbody.appendChild(tr);
            });
            table.appendChild(tbody);
            resultado.appendChild(table);
        }
    </script>
</head>
<body>
    <h1>Visualizar Dados da Viatura</h1>
    <input type="text" id="matricula" placeholder="MatrÃ­cula" />
    <button onclick="buscarDados()">Buscar Dados</button>
    <div id="resultado"></div>
</body>
